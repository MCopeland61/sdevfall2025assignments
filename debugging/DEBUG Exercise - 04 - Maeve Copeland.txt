// This pseudocode segment is intended to compute and display
// the cost of home ownership for any number of users
// The program ends when a user enters 0 for the mortgage payment
start
   Declarations
      num mortgagePayment
      num utilities
      num taxes
      num upkeep
      num total
   startup()
   while mortgagePayment <> 0 //changed for readibility
      MainLoop()
   endwhile
   output "End of Program" //adjusted a few things for memory reasons, read comment at bottom
   //finishUp()
stop

startUp()
   output "Enter your mortgage payment or 0 to quit"
   input mortgagePayment //changed to match the name of the actual variable being used
return

mainLoop()
   output "Enter utilities"
   input utilities
   output "Enter taxes"
   input taxes
   output "Enter amount for upkeep"
   input upkeep
   total = mortgagePayment + utilities + taxes + upkeep
   output "Total is ", total
   startUp() //called this due to it looping back to utilities which would lead to an infinite loop. in theory you could cut startUp() and incorporate it into the mainloop since mortgagePayment input has to be called again to be able to end the program
return
